            //自适应宽高

                myChart.style.width = window.innerWidth*proportion+'px';

            var myChart = echarts.init(myChart);

                    subtext: '响应数据'
                },
                tooltip: {
                    trigger: 'axis'
                },

                },

                        axisLine: {show: false},

                    }
                ],
                yAxis: [
                    {
                        type: 'value',
                        axisLabel: {formatter: '{value} 位'}
                    }
                ],
                series: [


                ]


            };

            // 获取类目,并且设置响应的File的下标
            for (var i = 0; i < meta.length; i++) {
                // 添加了类别，总共六个字段
                if (option.legend.data.length == 3) {
                    option.legend.data.push("")
                }
                var item_cur = meta[i].name
                if (item_cur !== "TOTAL_DATE") {
                    option.legend.data.push(item_cur)
                }
            }

            // 获取x轴上的时间数据、同时添加不同
            for (var i = 0; i < data.length; i++) {
                //获取每行数据的同时添加到数据
                option.xAxis[0].data.push(data[i][0])
                option.xAxis[1].data.push(data[i][0])
            }
            // 添加数据
            for (var i = 0; i < meta.length; i++) {
                if (meta[i].name !== "TOTAL_DATE") {
                    var series_obj = {}
                    series_obj.name = meta[i].name
                    series_obj.type = "bar"
                    series_obj.itemStyle = {
                        normal: {
                            //color: 'rgba(181,195,52,1)',
                            label: {show: false, textStyle: {color: '#27727B'}}
                        }
                    }
                    series_obj.data=[]
                    for (var j = 0; j < data.length; j++) {
                        series_obj.data.push(data[j][i])
                    }
                    option.series.push(series_obj)
                }
            }
            //alert(JSON.stringify(option))

            myChart.setOption(option);
            window.onresize = function () {
                myChartContainer();
                myChart.resize();
            };
        }
