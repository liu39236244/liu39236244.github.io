<!DOCTYPE html>
<html>
<head>

        // 配置所需要的图标路径


        function Print_Line(json_obj, div_id, proportion) {
            var data = json_obj.data
            var meta = json_obj.meta


            // 基于准备好的dom，初始化echarts实例
            var myChart = document.getElementById(div_id)
            //自适应宽高

                myChart.style.width = window.innerWidth * proportion + 'px';
                myChart.style.height = window.innerHeight * proportion + 'px';
            };

            option = {
                title: {
                    text: '标题',

                    orient: 'vertical',

            //
            var REG_USER_NUM_Array = 0

            var UV_Array = 0
            var STARTUP_NUM_Array = 0
            var KQMM_DOWNREPORT_Array = 0
            var KQMM_SOFTDOWN_Array = 0
            var All_Array = 0
            for (var data_index = 0; data_index < data.length; data_index++) {
                REG_USER_NUM_Array += data[data_index][1]
                PV_Array += data[data_index][2]
                UV_Array += data[data_index][3]

            }
            // 计算数据并写入
            // 每日注册量占比

            for (var category = 0; category < option.legend.data.length; category++) {
                var data_obj = {}
                var cate = option.legend.data[category]
                switch (cate) {
                    case "REG_USER_NUM" : {
                        data_obj.name=cate;
                        data_obj.value=REG_USER_NUM_Array/All_Array
                        break;
                    }
                    case "PV" : {
                        data_obj.name=cate;
                        data_obj.value=PV_Array/All_Array
                        break;
                    }
                    case "UV" : {
                        data_obj.name=cate;
                        data_obj.value=UV_Array/All_Array
                        break;
                    }
                    case "STARTUP_NUM" : {
                        data_obj.name=cate;
                        data_obj.value=STARTUP_NUM_Array/All_Array
                        break;
                    }
                    case "KQMM_DOWNREPORT" : {
                        data_obj.name=cate;
                        data_obj.value=KQMM_DOWNREPORT_Array/All_Array
                        break;
                    }
                    case "KQMM_SOFTDOWN" : {
                        data_obj.name=cate;
                        data_obj.value=KQMM_SOFTDOWN_Array/All_Array
                        break;
                    }
                }
                option.series[0].data.push(data_obj)
            }
            myChart.setOption(option);
        }
    </script>
